[row['NB起站'], row['NB迄站']].forEach(s => { if(s) stats.busStations[s] = (stats.busStations[s] || 0) + 1; }); } if(row['運具'] === '台鐵') { [row['TR起站'], row['TR迄站']].forEach(s => { if(s) stats.trainStations[s] = (stats.trainStations[s] || 0) + 1; }); } }); document.getElementById('total-savings').innerText = `NT$ ${Math.round(stats.total)}`; document.getElementById('yb-savings').innerText = Math.round(stats.yb); document.getElementById('bus-savings').innerText = Math.round(stats.bus); document.getElementById('train-savings').innerText = Math.round(stats.train); const sortedTrans = Object.entries(stats.transports).sort((a,b)=>b[1]-a[1]); document.getElementById('top-transport').innerText = sortedTrans[0] ? sortedTrans[0][0] : '--'; document.getElementById('yb-type-rank').innerText = ybTypeStr || '--'; document.getElementById('yb-bike-count').innerText = Object.keys(stats.ybBikes).length; const topBike = Object.entries(stats.ybBikes).sort((a,b)=>b[1]-a[1])[0]; document.getElementById('yb-top-bike').innerText = topBike ? topBike[0] : '--'; const topRoute = Object.entries(stats.ybRoutes).sort((a,b)=>b[1]-a[1])[0]; document.getElementById('yb-top-route').innerText = topRoute ? topRoute[0] : '無數據';
